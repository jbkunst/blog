Datos

Code
library(tidyverse)
library(scales)
library(patchwork)

data <- read_csv2(
  here::here("posts/2022-xx-xx-visualizacion-en-el-analisis-de-datos/data/2015.04_Subidas_paradero_mediahora_web.csv"),
  show_col_types = FALSE
  )

data <- data |> 
  mutate(subidas_laboral_promedio = as.numeric(subidas_laboral_promedio)) |> 
  filter(!str_detect(paraderosubida, "[0-9]+-[0-9]")) |> 
  filter(paraderosubida != "-") |> 
  filter(lubridate::hour(mediahora) > 0) 

data <- data |>
  mutate(mediahora = lubridate::as_datetime(mediahora))

lubridate::year(data$mediahora) <- 2015

data <- data |> 
  complete(
    paraderosubida, 
    mediahora, 
    fill = list(subidas_laboral_promedio = 0)
    )

Sobre equivocarse y las convenciones

A veces debemos seguir convenciones:

Code
dplazamaipu <- data |>
  filter(paraderosubida == "PLAZA MAIPU")

ggplot(dplazamaipu) +
  geom_point(aes(subidas_laboral_promedio, mediahora, color = paraderosubida), size = 1.5) +
  scale_y_datetime(date_labels = "%R") +
  scale_x_continuous(labels = comma) +
  scale_color_viridis_d(guide = "none") +
  labs(title = "Primer intento")
ggplot(dplazamaipu) +
  geom_path(aes(subidas_laboral_promedio, mediahora, color = paraderosubida), linewidth = 1.5) +
  scale_y_datetime(date_labels = "%R") +
  scale_x_continuous(labels = comma) +
  scale_color_viridis_d(guide = "none") +
  labs(title = "Segundo intento")

Combinar información

Importante observar patrones/relaciones en los datos.

Code
d1 <- data |>
  filter(paraderosubida %in% c("PLAZA MAIPU", "LAGUNA SUR"))

c <- d1 |>
  spread(paraderosubida, subidas_laboral_promedio) |>
  # filter(complete.cases(.)) |>
  select(-1) |>
  as.matrix() |>
  cor() |>
  as.vector() |>
  nth(2)

p1 <- ggplot(d1) +
  geom_line(
    aes(mediahora, subidas_laboral_promedio, color = paraderosubida, group = paraderosubida),
    linewidth = 1.2
  ) +
  scale_x_datetime(date_labels = "%R") +
  scale_y_continuous(label = scales::comma) +
  scale_color_viridis_d(name = NULL)
